    <script>
        function updateTime() {
            let now = new Date();
            document.getElementById("liveClock").innerText = now.toLocaleTimeString();
            document.getElementById("liveDate").innerText = now.toDateString();
        }
        setInterval(updateTime, 1000);
        updateTime();

        function calculateAge() {
            let birthdate = new Date(document.getElementById("birthdate").value);
            let now = new Date();
            let diff = now - birthdate;
            let ageYears = Math.floor(diff / (1000 * 60 * 60 * 24 * 365));
            let ageMonths = Math.floor(diff / (1000 * 60 * 60 * 24 * 30));
            let ageWeeks = Math.floor(diff / (1000 * 60 * 60 * 24 * 7));
            let ageDays = Math.floor(diff / (1000 * 60 * 60 * 24));
            let ageHours = Math.floor(diff / (1000 * 60 * 60));
            let ageMinutes = Math.floor(diff / (1000 * 60));
            let ageSeconds = Math.floor(diff / 1000);

            document.getElementById("years").innerText = ageYears + " Years";
            document.getElementById("months").innerText = ageMonths + " Months";
            document.getElementById("weeks").innerText = ageWeeks + " Weeks";
            document.getElementById("days").innerText = ageDays + " Days";
            document.getElementById("hours").innerText = ageHours + " Hours";
            document.getElementById("minutes2").innerText = ageMinutes + " Minutes";
            document.getElementById("seconds2").innerText = ageSeconds + " Seconds";
        }
    </script>
